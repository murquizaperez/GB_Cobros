function formatNumber(num) {
  return (parseFloat(num) || 0).toFixed(2);
}

function formatCantidad(num) {
  const n = parseFloat(num) || 0;
  return Number.isInteger(n) ? String(n) : n.toFixed(3);
}

function escapeHtml(value) {
  return String(value || '')
    .replace(/&/g, '&')
    .replace(//g, '>')
    .replace(/"/g, '"')
    .replace(/'/g, ''');
}

function domId(value) {
  return String(value || '').replace(/[^a-zA-Z0-9_-]/g, '_');
}

function showToast(message, type = 'info') {
  const colors = { success: 'bg-success', error: 'bg-danger', warning: 'bg-warning', info: 'bg-info' };
  const container = document.getElementById('toast-container');
  const toast = document.createElement('div');
  toast.className = `toast px-4 py-3 rounded-lg text-white ${colors[type] || colors.info}`;
  toast.textContent = message;
  container.appendChild(toast);
  setTimeout(() => toast.remove(), 3000);
}

function closeModal(id) {
  document.getElementById(id).classList.remove('active');
}

function startButtonLoading(key, button, text = 'Procesando...') {
  if (operationLocks[key]) return false;
  operationLocks[key] = true;
  if (button) {
    button.dataset.originalHtml = button.innerHTML;
    button.disabled = true;
    button.classList.add('loading');
    button.innerHTML = `${text}`;
  }
  return true;
}

function stopButtonLoading(key, button) {
  operationLocks[key] = false;
  if (button) {
    button.disabled = false;
    button.classList.remove('loading');
    if (button.dataset.originalHtml) button.innerHTML = button.dataset.originalHtml;
  }
}

function runButtonTask(key, button, text, task) {
  if (!startButtonLoading(key, button, text)) return Promise.resolve(null);
  return Promise.resolve()
    .then(task)
    .catch(err => {
      console.error(err);
      showToast('Error de operación. Intenta nuevamente.', 'error');
    })
    .finally(() => stopButtonLoading(key, button));
}

function navigateTo(page) {
  currentPage = page;
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('bg-secondary'));
  
  const pageEl = document.getElementById('page-' + page);
  if (pageEl) pageEl.classList.add('active');
  
  const link = document.querySelector(`[data-page="${page}"]`);
  if (link) link.classList.add('bg-secondary');
  
  const titles = {
    dashboard: 'Dashboard', produccion: 'Produccion y Lotes', ingredientes: 'Ingredientes', productos: 'Productos',
    precios: 'Precios y Margenes', caja: 'Caja', pos: 'Punto de Venta',
    reportes: 'Reportes', pedidos: 'Pedidos', clientes: 'Clientes', config: 'Configuracion'
  };
  document.getElementById('page-title').textContent = titles[page] || page;
  
  loadPageData(page);
}

function calcularCosto(cantidad, unidadReceta, costoBase, unidadBase) {
  let qty = parseFloat(cantidad) || 0;
  let baseU = String(unidadBase).toLowerCase().trim();
  let recU = String(unidadReceta).toLowerCase().trim();
  let factor = 1;
  
  // Conversor inteligente de unidades
  if (baseU === 'kg' && recU === 'g') factor = 0.001;
  else if (baseU === 'g' && recU === 'kg') factor = 1000;
  else if (baseU === 'l' && recU === 'ml') factor = 0.001;
  else if (baseU === 'ml' && recU === 'l') factor = 1000;
  
  return (qty * factor) * (parseFloat(costoBase) || 0);
}

// NUEVA FUNCIÓN INTELIGENTE: Calcula el costo de cualquier producto instantáneamente
function getProductCostoTotal(prod) {
  // Si no tenemos ingredientes en memoria o el producto no tiene receta, devolver su valor directo
  if (!ingredientesData || ingredientesData.length === 0) return parseFloat(prod.Costo_Total) || 0;
  
  let totalCalculado = 0;
  let tieneRecetaValida = false;
  
  if (prod.receta && prod.receta.length > 0) {
    prod.receta.forEach(r => {
      // Soporta propiedades enviadas por el script o mapeadas localmente
      const ingId = r.Ingrediente_ID || r.ingredienteId; 
      const ing = ingredientesData.find(i => String(i.ID) === String(ingId));
      
      if (ing) {
        tieneRecetaValida = true;
        const baseU = String(ing.Unidad_Medida).toLowerCase().trim();
        const recU = r.Unidad_Medida ? String(r.Unidad_Medida).toLowerCase().trim() : baseU;
        const qty = parseFloat(r.Cantidad || r.cantidad) || 0;
        const baseCost = parseFloat(ing.Costo_Por_Unidad) || 0;
        
        totalCalculado += calcularCosto(qty, recU, baseCost, baseU);
      }
    });
  }
  
  return tieneRecetaValida ? totalCalculado : (parseFloat(prod.Costo_Total) || 0);
}

  if (ingredientesData.length === 0) {

  // Si editamos un producto existente, recalcular el costo de cada linea dinamicamente
  if (prod && prod.receta && prod.receta.length > 0) {
    recetaItems = prod.receta.map(r => {
      const ing = ingredientesData.find(i => String(i.ID) === String(r.Ingrediente_ID));
      const baseU = ing ? String(ing.Unidad_Medida).toLowerCase().trim() : '';

      return {
        ingredienteId: String(r.Ingrediente_ID),
        cantidad: qty,
        unidadMedida: recU,
        costoLinea: calcularCosto(qty, recU, baseCost, baseU)

function updateRecetaRow(idx, field, value) {
  recetaItems[idx][field] = value;
  const ing = ingredientesData.find(i => String(i.ID) === String(recetaItems[idx].ingredienteId));
  
  if (ing) {
    // Si cambio el ingrediente, asignar su unidad base por defecto
    if (field === 'ingredienteId') {
      recetaItems[idx].unidadMedida = String(ing.Unidad_Medida).toLowerCase().trim();
    }
    
    // Recalcular el costo exacto con la formula
    recetaItems[idx].costoLinea = calcularCosto(
      recetaItems[idx].cantidad,
      recetaItems[idx].unidadMedida,
      ing.Costo_Por_Unidad,
      ing.Unidad_Medida
    );
  } else {
    recetaItems[idx].costoLinea = 0;
  }
  
  renderReceta();
  calcCostoTotal();
}

function removeRecetaRow(idx) {
  recetaItems.splice(idx, 1);
  renderReceta();
  calcCostoTotal();
}

function calcCostoTotal() {
  const total = recetaItems.reduce((sum, item) => sum + (item.costoLinea || 0), 0);
  document.getElementById('prod-costo-total').textContent = '$' + formatNumber(total);
  return total;
}

function getOrderedPOSProducts(data) {
  const productos = [...(data || [])];
  if (!posProductOrder.length) return productos;
  const orderMap = {};
  posProductOrder.forEach((id, idx) => orderMap[String(id)] = idx);
  return productos.sort((a, b) => {
    const ia = orderMap[String(a.ID)];
    const ib = orderMap[String(b.ID)];
    if (ia === undefined && ib === undefined) return 0;
    if (ia === undefined) return 1;
    if (ib === undefined) return -1;
    return ia - ib;
  });
}

function handlePOSDrop(event, targetId) {
  event.preventDefault();
  const sourceId = event.dataTransfer.getData('text/plain');
  event.currentTarget.classList.remove('drag-over');
  if (!sourceId || String(sourceId) === String(targetId)) return;

  const activeIds = productosData.filter(p => p.Activo).map(p => String(p.ID));
  const currentOrder = posProductOrder.length ? [...posProductOrder] : activeIds;
  activeIds.forEach(id => { if (!currentOrder.includes(id)) currentOrder.push(id); });

  const withoutSource = currentOrder.filter(id => String(id) !== String(sourceId));
  const targetIndex = withoutSource.findIndex(id => String(id) === String(targetId));
  withoutSource.splice(targetIndex < 0 ? withoutSource.length : targetIndex, 0, String(sourceId));
  posProductOrder = withoutSource;
  savePOSProductOrder();
  filterPOSProducts(document.getElementById('pos-search')?.value || '');
}

function handlePOSDragEnd(event) {
  event.currentTarget.classList.remove('dragging', 'drag-over');
}

function finalizarVenta(button = null) {
  if (carritoVenta.length === 0) return;
  const items = carritoVenta.map(item => ({ productoId: item.id, cantidad: item.cantidad, precioUnitario: item.precio }));
  const total = carritoVenta.reduce((sum, item) => sum + (item.precio * item.cantidad), 0);
  const atendidoPor = cajaActual && cajaActual.encargadoTurno ? cajaActual.encargadoTurno : '';
  const esRegalo = document.getElementById('pos-es-regalo')?.checked || false;
  const medioPago = getMedioPagoPOS();
  const clienteNombre = (document.getElementById('pos-cliente')?.value || '').trim();

  if (!esRegalo && !medioPago) {
    showToast('Indica el tipo de pago', 'warning');
    return;
  }
  
  return runButtonTask('finalizarVenta', button, esRegalo ? 'Registrando...' : 'Cobrando...', () => api('registrarVenta', items, total, {
    atendidoPor: atendidoPor,
    medioPago: medioPago,
    clienteNombre: clienteNombre,
    esRegalo: esRegalo
  }).then(result => {
    if (result.success) {
      showToast(esRegalo ? 'Regalo registrado' : 'Venta registrada: $' + formatNumber(result.totalCobrado || result.total), 'success');
      
      // Descontar del stock en la UI optimista
      carritoVenta.forEach(item => {
          const pIdx = productosData.findIndex(x => String(x.ID) === String(item.id));
          if(pIdx > -1) {
              productosData[pIdx].Stock_Actual = Math.max(0, (parseFloat(productosData[pIdx].Stock_Actual) || 0) - item.cantidad);
              api('updateProducto', item.id, { stockActual: productosData[pIdx].Stock_Actual });
          }
      });
      
      const ventaNueva = {
        ID: result.id,
        Numero_Venta: result.numeroVenta,
        Cliente_Nombre: result.clienteNombre,
        Medio_Pago: result.medioPago,
        Es_Regalo: result.esRegalo,
        Total: result.total,
        Total_Cobrado: result.totalCobrado,
        items: carritoVenta.map(item => ({ Producto_Nombre: item.nombre, Cantidad: item.cantidad, Subtotal: item.precio * item.cantidad }))
      };
      ventasCajaData = [ventaNueva, ...ventasCajaData];
      carritoVenta = [];
      renderCarrito();
      document.getElementById('pos-cliente').value = '';
      document.getElementById('pos-es-regalo').checked = false;
      toggleRegaloPOS();
      renderVentasCajaList(ventasCajaData);
      if (cajaActual) {
        cajaActual.totalVentas = (parseFloat(cajaActual.totalVentas) || 0) + (parseFloat(result.totalCobrado) || 0);
        cajaActual.cantidadVentas = (parseInt(cajaActual.cantidadVentas, 10) || 0) + 1;
      }
      renderPOSProducts(getOrderedPOSProducts(productosData.filter(p => p.Activo && p.Nombre.toLowerCase().includes((document.getElementById('pos-search')?.value || '').toLowerCase()))));
      api('getMetodosPago').then(data => { metodosPagoData = data || metodosPagoData; });
      api('getCajaActual').then(d => { cajaActual = d; updateCajaStatus(d && d.abierta); });
    } else {
      showToast(result.error || 'Error al registrar venta', 'error');
    }
  }));
}

function playPedidoAlert() {
  try {
    const AudioContextClass = window.AudioContext || window.webkitAudioContext;
    if (!AudioContextClass) return;
    const ctx = new AudioContextClass();
    const gain = ctx.createGain();
    gain.gain.value = 0.12;
    gain.connect(ctx.destination);

    [0, 180].forEach(delay => {
      const osc = ctx.createOscillator();
      osc.type = 'sine';
      osc.frequency.value = 880;
      osc.connect(gain);
      osc.start(ctx.currentTime + delay / 1000);
      osc.stop(ctx.currentTime + delay / 1000 + 0.14);
    });
  } catch (err) {
    console.log('Audio no disponible', err);
  }
}

function previewRecetaTeorica() {
  const prodId = document.getElementById('prod-select-producto').value;
  const cant = parseFloat(document.getElementById('prod-cant-teorica').value) || 0;
  const container = document.getElementById('prod-preview-receta');
  const costoEl = document.getElementById('prod-costo-estimado');
  
  costoEl.dataset.valor = "0";

  if(!prodId || cant <= 0) { 
      container.innerHTML = 'Ingrese datos validos.'; 
      costoEl.textContent='$0.00'; 
      return; 
  }
  
  const prod = productosData.find(x => String(x.ID) === String(prodId));
  if(!prod || !prod.receta || prod.receta.length === 0) { 
      container.innerHTML = 'El producto no tiene receta cargada.'; 
      costoEl.textContent='$0.00';

  let costoTotal = 0;
  container.innerHTML = prod.receta.map(r => {
    const ingId = r.Ingrediente_ID || r.ingredienteId;
    const ing = ingredientesData.find(x => String(x.ID) === String(ingId));

    const cantTeoricaParaReceta = (parseFloat(r.Cantidad || r.cantidad) || 0) * cant;

`;
  }).join('');
  
  costoEl.textContent = '$' + formatNumber(costoTotal); 
  costoEl.dataset.valor = costoTotal;
}

function submitIniciarProduccion(e) {
  e.preventDefault();
  try {
      const prodId = document.getElementById('prod-select-producto').value;
      const cant = parseFloat(document.getElementById('prod-cant-teorica').value) || 0;
      
      // Forzar calculo si no lo habia hecho automatico
      let costoTeo = parseFloat(document.getElementById('prod-costo-estimado').dataset.valor) || 0;
      if (costoTeo === 0) {
         previewRecetaTeorica();
         costoTeo = parseFloat(document.getElementById('prod-costo-estimado').dataset.valor) || 0;
      }
      
      const empleado = document.getElementById('prod-empleado').value;
      
      const prod = productosData.find(x => String(x.ID) === String(prodId));
      if(!prod || !prod.receta || prod.receta.length === 0) {
          showToast('El producto no tiene receta válida.', 'warning');
          return;
      }

      // Filtramos para asegurar que el ingrediente aún existe y prevenir errores de lectura "undefined"
      const ingredientesValidos = prod.receta.filter(r => {
          const ingId = r.Ingrediente_ID || r.ingredienteId;
          return ingredientesData.some(x => String(x.ID) === String(ingId));
      });

      if(ingredientesValidos.length === 0) {
          showToast('Los ingredientes de la receta no existen en el inventario.', 'warning');
          return;
      }

      const detallesTeoricos = {
        ingredientes: ingredientesValidos.map(r => {
          const ingId = r.Ingrediente_ID || r.ingredienteId;
          const ing = ingredientesData.find(x => String(x.ID) === String(ingId));
          const cantTotal = (parseFloat(r.Cantidad || r.cantidad) || 0) * cant;
          const recU = r.Unidad_Medida ? String(r.Unidad_Medida).toLowerCase().trim() : String(ing.Unidad_Medida).toLowerCase().trim();
          return { id: ing.ID, nombre: ing.Nombre, unidad: recU, cantidadTeorica: cantTotal, costoU: ing.Costo_Por_Unidad, unidadBase: ing.Unidad_Medida };
        })
      };

      const btn = e.submitter; 
      if(btn) { btn.disabled = true; btn.textContent = 'Iniciando...'; }
      
      api('iniciarProduccion', { productoId: prodId, empleado: empleado, cantidadTeorica: cant, costoTeorico: costoTeo, detallesTeoricos: detallesTeoricos }).then(res => {
        if(res.success) { 
          showToast('Lote iniciado: ' + res.lote, 'success'); 
          closeModal('modal-iniciar-produccion'); 

          // ACTUALIZACION OPTIMISTA: Forzar que aparezca inmediatamente en pantalla
          produccionesData.push({
              ID: res.id,
              Lote: res.lote || 'LOTE-NUEVO',
              Fecha_Inicio: res.fechaInicio || new Date().toISOString(),
              Estado: 'EN_PROCESO',
              Producto_ID: prodId,
              Empleado: empleado,
              Cantidad_Teorica: cant,
              Costo_Teorico: costoTeo,
              Detalles: detallesTeoricos
          });
          renderProducciones();

          api('getProducciones').then(data => { 
             if(data && data.length > 0) produccionesData = data;
             renderProducciones(); 
          }).catch(e => console.log('Silenced error getProducciones', e)); 

        } else {
          showToast(res.error, 'error');
        }
      }).finally(() => { 
          if(btn) { btn.disabled = false; btn.textContent = 'Generar Lote e Iniciar'; }
      });
  } catch (err) {
      console.error(err);
      showToast('Error al procesar la receta.', 'error');
  }
}

      const m = isNaN(diffMs) ? 0 : Math.floor((diffMs % 3600000) / 60000);
      const tiempoFinal = `${h}h ${m}m`;
      
      const ings = (produccionActivaSelected.Detalles && produccionActivaSelected.Detalles.ingredientes) ? produccionActivaSelected.Detalles.ingredientes : [];
      const ingredientesUsados = ings.map((ing, idx) => ({ 
        id: ing.id, nombre: ing.nombre, cantidadTeorica: ing.cantidadTeorica, 
        cantidadUsada: parseFloat(document.getElementById(`fin-ing-real-${idx}`).value) || 0, unidad: ing.unidad 
      }));
      
      const payload = { 
        cantidadReal: cantReal, costoReal: costoReal, merma: notas, 
        desviacion: cantReal - produccionActivaSelected.Cantidad_Teorica, 
        detallesReales: { 
            ingredientesUsados: ingredientesUsados, 
            costoTeoricoOriginal: produccionActivaSelected.Costo_Teorico,
            tiempoTranscurrido: tiempoFinal // Se guarda para trazabilidad permanente
        } 
      };
      
      const btn = e.submitter; 
      if(btn) { btn.disabled = true; btn.innerHTML = 'Procesando...'; }
      
      api('finalizarProduccion', id, payload).then(res => {
        if(res.success) {
          showToast('Lote finalizado. Stock descontado y Producto actualizado.', 'success'); 
          closeModal('modal-finalizar-produccion');

          // ACTUALIZACION OPTIMISTA DEL LOTE EN PANTALLA
          if(produccionActivaSelected) {
              produccionActivaSelected.Estado = 'FINALIZADA';
              produccionActivaSelected.Fecha_Fin = new Date().toISOString();
              produccionActivaSelected.Cantidad_Real = cantReal;
              produccionActivaSelected.Costo_Real = costoReal;
              produccionActivaSelected.Merma_Desviacion = notas;
              if(!produccionActivaSelected.Detalles) produccionActivaSelected.Detalles = {};
              produccionActivaSelected.Detalles.tiempoTranscurrido = payload.detallesReales.tiempoTranscurrido;
          }
          
          // ACTUALIZACION OPTIMISTA DE STOCK PARA EL PRODUCTO
          const prodId = produccionActivaSelected.Producto_ID || produccionActivaSelected.productoId || produccionActivaSelected.producto_id;
          const prodIndex = productosData.findIndex(x => String(x.ID) === String(prodId));
          if (prodIndex > -1) {
              productosData[prodIndex].Stock_Actual = (parseFloat(productosData[prodIndex].Stock_Actual) || 0) + cantReal;
              // Informamos al backend que aumente el stock, incluso si el backend no lo procesa ahora, lo mantenemos en la UI
              api('updateProducto', prodId, { stockActual: productosData[prodIndex].Stock_Actual });
          }

          renderProducciones();

          api('getInitialData').then(data => { 
            if(data) { 
              ingredientesData = data.ingredientes||[]; 
              if(data.producciones && data.producciones.length > 0) produccionesData = data.producciones; 
              
              // Si el backend devuelve los productos actualizados, los usamos, sino conservamos el stock optimista
              if(data.productos && data.productos.length > 0) {
                 data.productos.forEach(newP => {
                    const oldP = productosData.find(p => String(p.ID) === String(newP.ID));
                    // Si el backend no retorna Stock_Actual (porque le falta la columna en Sheets), conservamos el valor local
                    if (oldP && oldP.Stock_Actual !== undefined && newP.Stock_Actual === undefined) {
                        newP.Stock_Actual = oldP.Stock_Actual;
                    }
                 });
                 productosData = data.productos;
              }
              
              renderProducciones(); 
              if(currentPage === 'ingredientes') renderIngredientes(ingredientesData);
              if(currentPage === 'productos') renderProductos(productosData);
            }
          }).catch(e => console.log('Error recargando en background', e));
        } else showToast(res.error, 'error');
      }).finally(() => { 
          if(btn) { btn.disabled = false; btn.innerHTML = 'Descontar Stock y Finalizar'; }
      });
  } catch (err) {
      console.error(err);
      showToast('Error al cerrar el lote de producción', 'error');
  }
}
